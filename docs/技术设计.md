## 核心服务设计

### 1. 定位服务 (LocationService)

```typescript
// services/LocationService.ets
import geoLocationManager from '@ohos.geoLocationManager';
import backgroundTaskManager from '@ohos.backgroundTaskManager';
import { BusinessError } from '@ohos.base';

export class LocationService {
  private static instance: LocationService;
  private locationCallback: ((location: geoLocationManager.Location) => void) | null = null;
  private isTracking: boolean = false;
  private cachedLocation: geoLocationManager.Location | null = null;

  static getInstance(): LocationService {
    if (!LocationService.instance) {
      LocationService.instance = new LocationService();
    }
    return LocationService.instance;
  }

  // 获取当前位置（单次）
  async getCurrentLocation(): Promise<geoLocationManager.Location | null> {
    try {
      // 先返回缓存的位置（如果有且在5分钟内）
      if (this.cachedLocation && 
          Date.now() - this.cachedLocation.timestamp < 5 * 60 * 1000) {
        return this.cachedLocation;
      }

      const requestInfo: geoLocationManager.SingleLocationRequest = {
        locatingTimeoutMs: 15000,
        locatinigMaxAge: 60 * 1000  // 1分钟内的缓存
      };

      return new Promise((resolve) => {
        geoLocationManager.getCurrentLocation(requestInfo, (err, location) => {
          if (err || !location) {
            console.error('LocationService', 'getCurrentLocation failed', err);
            resolve(this.cachedLocation);
          } else {
            this.cachedLocation = location;
            resolve(location);
          }
        });
      });
    } catch (error) {
      console.error('LocationService', 'getCurrentLocation error', error);
      return this.cachedLocation;
    }
  }

  // 开始持续定位
  async startContinuousTracking(
    callback: (location: geoLocationManager.Location) => void
  ): Promise<boolean> {
    if (this.isTracking) {
      return true;
    }

    try {
      // 申请后台长时任务
      await this.requestBackgroundTask();

      this.locationCallback = callback;
      
      // 配置定位参数
      const requestInfo: geoLocationManager.LocationRequest = {
        priority: geoLocationManager.LocationRequestPriority.ACCURACY,
        timeInterval: 60,  // 60秒 = 1分钟
        distanceInterval: 0
      };

      geoLocationManager.on('locationChange', requestInfo, (location: geoLocationManager.Location) => {
        if (location) {
          this.cachedLocation = location;
          if (this.locationCallback) {
            this.locationCallback(location);
          }
        }
      });

      this.isTracking = true;
      console.info('LocationService', 'startContinuousTracking success');
      return true;
    } catch (error) {
      console.error('LocationService', 'startContinuousTracking error', error);
      return false;
    }
  }

  // 停止持续定位
  stopContinuousTracking(): void {
    if (!this.isTracking || !this.locationCallback) {
      return;
    }

    try {
      geoLocationManager.off('locationChange', this.locationCallback);
      this.locationCallback = null;
      this.isTracking = false;
      
      // 释放后台任务
      this.releaseBackgroundTask();
      
      console.info('LocationService', 'stopContinuousTracking success');
    } catch (error) {
      console.error('LocationService', 'stopContinuousTracking error', error);
    }
  }

  // 申请后台长时任务
  private async requestBackgroundTask(): Promise<void> {
    try {
      backgroundTaskManager.requestSuspendDelay('travel_tracker', () => {
        console.info('LocationService', 'background task expired');
      });
    } catch (error) {
      console.error('LocationService', 'requestBackgroundTask error', error);
    }
  }

  // 释放后台任务
  private releaseBackgroundTask(): void {
    try {
      backgroundTaskManager.cancelSuspendDelay('travel_tracker');
    } catch (error) {
      console.error('LocationService', 'releaseBackgroundTask error', error);
    }
  }

  // 检查定位权限
  async checkLocationPermission(): Promise<boolean> {
    // 实现权限检查逻辑
    return true;
  }
}
```

### 2. 记录服务 (RecordService)

```typescript
// services/RecordService.ets
import { LocationService } from './LocationService';
import { DatabaseService } from './DatabaseService';
import { RecordMode, RecordSegment, TrackPoint, StayPoint } from '../models';

export class RecordService {
  private static instance: RecordService;
  private locationService: LocationService;
  private dbService: DatabaseService;
  
  private currentSegment: RecordSegment | null = null;
  private lastTrackPoint: TrackPoint | null = null;
  private currentStayPoint: StayPoint | null = null;
  
  private isRecording: boolean = false;
  private recordMode: RecordMode = RecordMode.DRIVING;

  // 停留点合并阈值（米）
  private readonly STAY_DISTANCE_THRESHOLD = 50;
  
  static getInstance(): RecordService {
    if (!RecordService.instance) {
      RecordService.instance = new RecordService();
    }
    return RecordService.instance;
  }

  constructor() {
    this.locationService = LocationService.getInstance();
    this.dbService = DatabaseService.getInstance();
  }

  // 开始记录
  async startRecord(tripId: string, mode: RecordMode): Promise<RecordSegment | null> {
    if (this.isRecording) {
      return null;
    }

    // 创建新的记录段
    this.currentSegment = {
      id: this.generateId(),
      tripId,
      startTime: Date.now(),
      endTime: null,
      mode,
      totalDistance: 0,
      duration: 0,
      trackPointCount: 0,
      stayPointCount: 0
    };

    this.recordMode = mode;
    this.lastTrackPoint = null;
    this.currentStayPoint = null;
    
    // 保存到数据库
    await this.dbService.insertRecordSegment(this.currentSegment);
    
    // 开始持续定位
    await this.locationService.startContinuousTracking((location) => {
      this.onLocationUpdate(location);
    });

    this.isRecording = true;
    return this.currentSegment;
  }

  // 位置更新回调
  private async onLocationUpdate(location: geoLocationManager.Location): Promise<void> {
    if (!this.isRecording || !this.currentSegment) {
      return;
    }

    const trackPoint: TrackPoint = {
      id: this.generateId(),
      segmentId: this.currentSegment.id,
      latitude: location.latitude,
      longitude: location.longitude,
      altitude: location.altitude,
      accuracy: location.accuracy,
      timestamp: Date.now(),
      speed: location.speed
    };

    // 自动合并逻辑：检测是否为停留点
    if (this.lastTrackPoint) {
      const distance = this.calculateDistance(
        trackPoint.latitude, trackPoint.longitude,
        this.lastTrackPoint.latitude, this.lastTrackPoint.longitude
      );

      if (distance < this.STAY_DISTANCE_THRESHOLD) {
        // 距离小于阈值，合并到当前停留点
        await this.mergeToStayPoint(trackPoint);
      } else {
        // 距离大于阈值，保存上一个停留点（如果有），创建新点
        if (this.currentStayPoint) {
          await this.saveStayPoint();
        }
        
        // 保存轨迹点
        await this.saveTrackPoint(trackPoint);
        this.lastTrackPoint = trackPoint;
      }
    } else {
      // 第一个点，直接保存
      await this.saveTrackPoint(trackPoint);
      this.lastTrackPoint = trackPoint;
    }

    // 更新记录段统计
    await this.updateSegmentStats();
  }

  // 合并到停留点
  private async mergeToStayPoint(point: TrackPoint): Promise<void> {
    if (!this.currentStayPoint) {
      // 创建新的停留点
      this.currentStayPoint = {
        id: this.generateId(),
        segmentId: this.currentSegment!.id,
        latitude: point.latitude,
        longitude: point.longitude,
        altitude: point.altitude,
        startTime: point.timestamp,
        endTime: point.timestamp,
        duration: 0,
        mergedPointCount: 1,
        address: null
      };
    } else {
      // 更新现有停留点
      this.currentStayPoint.endTime = point.timestamp;
      this.currentStayPoint.duration = 
        (this.currentStayPoint.endTime - this.currentStayPoint.startTime) / 1000;
      this.currentStayPoint.mergedPointCount++;
    }

    // 不立即保存，等下次位置更新或结束时保存
  }

  // 保存停留点
  private async saveStayPoint(): Promise<void> {
    if (!this.currentStayPoint) {
      return;
    }

    // 获取地址
    try {
      this.currentStayPoint.address = await this.getAddress(
        this.currentStayPoint.latitude,
        this.currentStayPoint.longitude
      );
    } catch (error) {
      console.warn('RecordService', 'getAddress failed', error);
    }

    await this.dbService.insertStayPoint(this.currentStayPoint);
    
    if (this.currentSegment) {
      this.currentSegment.stayPointCount++;
    }
    
    this.currentStayPoint = null;
  }

  // 保存轨迹点
  private async saveTrackPoint(point: TrackPoint): Promise<void> {
    await this.dbService.insertTrackPoint(point);
    
    if (this.currentSegment) {
      this.currentSegment.trackPointCount++;
    }
  }

  // 停止记录
  async stopRecord(): Promise<void> {
    if (!this.isRecording || !this.currentSegment) {
      return;
    }

    // 停止定位
    this.locationService.stopContinuousTracking();

    // 保存最后的停留点
    if (this.currentStayPoint) {
      await this.saveStayPoint();
    }

    // 更新记录段结束时间
    this.currentSegment.endTime = Date.now();
    this.currentSegment.duration = 
      (this.currentSegment.endTime - this.currentSegment.startTime) / 1000;
    
    await this.dbService.updateRecordSegment(this.currentSegment);

    this.isRecording = false;
    this.currentSegment = null;
    this.lastTrackPoint = null;
  }

  // 计算两点距离（Haversine公式）
  private calculateDistance(
    lat1: number, lon1: number,
    lat2: number, lon2: number
  ): number {
    const R = 6371000; // 地球半径（米）
    const dLat = this.toRad(lat2 - lat1);
    const dLon = this.toRad(lon2 - lon1);
    const a = Math.sin(dLat / 2) * Math.sin(dLat / 2) +
              Math.cos(this.toRad(lat1)) * Math.cos(this.toRad(lat2)) *
              Math.sin(dLon / 2) * Math.sin(dLon / 2);
    const c = 2 * Math.atan2(Math.sqrt(a), Math.sqrt(1 - a));
    return R * c;
  }

  private toRad(deg: number): number {
```

```typescript
export interface ApiRequest {

  path: string;
  params?: Record<string, any>;
  body?: any;
  requireAuth?: boolean;
}

export interface ApiResponse<T> {
  code: number;
  data: T | null;
  message: string;
  timestamp: number;
}

export class ApiService {
  private static instance: ApiService;
  private authService: AuthService;
  
  // API配置
  private readonly BASE_URL = 'https://api.traveltracker.com';
  private readonly API_VERSION = 'v1';

  // 请求超时
  private readonly TIMEOUT = 30000;

  // 重试次数
  private readonly MAX_RETRIES = 3;

  static getInstance(): ApiService {
    if (!ApiService.instance) {
      ApiService.instance = new ApiService();
    }
    return ApiService.instance;
  }

  constructor() {
    this.authService = AuthService.getInstance();
  }

  // 通用请求方法
  async request<T>(request: ApiRequest): Promise<ApiResponse<T>> {
    // 1. 构建完整URL
    const url = `${this.BASE_URL}/${this.API_VERSION}${request.path}`;
    
    // 2. 添加公共参数
    const timestamp = Date.now();
    const nonce = this.generateNonce();
    
    // 3. 添加认证信息
    let headers: Record<string, string> = {
      'Content-Type': 'application/json',
      'X-Timestamp': timestamp.toString(),
      'X-Nonce': nonce,
      'X-Api-Version': this.API_VERSION
    };

    if (request.requireAuth !== false) {
      const user = this.authService.getCurrentUser();
      if (!user) {
        throw new Error('未登录');
      }
      headers['Authorization'] = `Bearer ${user.accessToken}`;
    }

    // 4. 生成请求签名
    const signature = this.generateSignature({
      method: request.method,
      path: request.path,
      timestamp,
      nonce,
      body: request.body
    });
    headers['X-Signature'] = signature;

    // 5. 发送请求（带重试）
    return this.sendRequestWithRetry<T>(url, {
      method: request.method,
      headers,
      params: request.params,
      body: request.body
    });
  }

  private generateSignature(params: {
    method: string;
    path: string;
    timestamp: number;
    nonce: string;
    body?: any;
  }): string {
    // 1. 构建签名字符串
    const bodyStr = params.body ? JSON.stringify(params.body) : '';
    const signStr = [
      params.method.toUpperCase(),
      params.path,
      params.timestamp.toString(),
      params.nonce,
      bodyStr
    ].join('\n');

    // 2. HMAC-SHA256签名
    const encoder = new crypto.framework.Cipher();
    const key = crypto.createSymKeyGenerator('HMAC_SHA256')
      .convertKey({ data: this.stringToUint8Array(this.signSecret) });
    
    const signature = crypto.createMac('HMAC_SHA256')
      .init(key)
      .update({ data: this.stringToUint8Array(signStr) })
      .doFinal();

    // 3. Base64编码
    return this.uint8ArrayToBase64(signature.data);
  }

  private async sendRequestWithRetry<T>(
    url: string,
    options: RequestInit,
    retryCount: number = 0
  ): Promise<ApiResponse<T>> {
    try {
      const response = await fetch(url, {
        ...options,
        signal: AbortSignal.timeout(this.TIMEOUT)
      });

      // 处理401未授权
      if (response.status === 401) {
        // 尝试刷新Token

        if (refreshed && retryCount < this.MAX_RETRIES) {

          return this.sendRequestWithRetry<T>(url, options, retryCount + 1);
        } else {
          // 需要重新登录

        }

      }
      throw error;
    }
  }

    if (!serverSignature) {
```

```typescript
export class DataEncryptor {
  private readonly ALGORITHM = 'AES256-GCM';
  private key: cryptoFramework.SymKey | null = null;

  // 初始化密钥（从服务器获取或本地生成）
  async init(userKey: string): Promise<void> {
    const keyGenerator = crypto.createSymKeyGenerator('AES256');
    this.key = await keyGenerator.convertKey({
      data: this.stringToUint8Array(userKey)
    });
  }

  // 加密
  async encrypt(plainText: string): Promise<string> {
    if (!this.key) {
      throw new Error('Key not initialized');
    }

    const cipher = crypto.createCipher(this.ALGORITHM);
    await cipher.init(cryptoFramework.CryptoMode.ENCRYPT_MODE, this.key);
    
    const encrypted = await cipher.doFinal({
      data: this.stringToUint8Array(plainText)
    });

    // 返回：iv + ciphertext + authTag (Base64)
    return this.uint8ArrayToBase64(encrypted.data);
  }

  // 解密
  async decrypt(cipherText: string): Promise<string> {
    if (!this.key) {
      throw new Error('Key not initialized');
    }

    const cipher = crypto.createCipher(this.ALGORITHM);
    await cipher.init(cryptoFramework.CryptoMode.DECRYPT_MODE, this.key);
    
    const decrypted = await cipher.doFinal({
      data: this.base64ToUint8Array(cipherText)
    });

    return this.uint8ArrayToString(decrypted.data);
  }

  private stringToUint8Array(str: string): Uint8Array {
    return new TextEncoder().encode(str);
  }

  private uint8ArrayToString(bytes: Uint8Array): string {
    return new TextDecoder().decode(bytes);
  }

  private uint8ArrayToBase64(bytes: Uint8Array): string {
    return btoa(String.fromCharCode(...bytes));
  }

  private base64ToUint8Array(base64: string): Uint8Array {
    const binary = atob(base64);
    const bytes = new Uint8Array(binary.length);
    for (let i = 0; i < binary.length; i++) {
      bytes[i] = binary.charCodeAt(i);
    }
    return bytes;
  }
}
```

```typescript
// 请求时间戳校验
class ReplayProtector {
  private readonly MAX_DELAY = 5 * 60 * 1000; // 5分钟
  private usedNonces: Set<string> = new Set();
  private cleanupInterval: number | null = null;

  // 验证请求
  validateRequest(timestamp: number, nonce: string): boolean {
    const now = Date.now();
    
    // 1. 检查时间戳（防重放）
    if (Math.abs(now - timestamp) > this.MAX_DELAY) {
      return false;
    }

    // 2. 检查Nonce（防重复）
    if (this.usedNonces.has(nonce)) {
      return false;
    }

    // 3. 记录Nonce
    this.usedNonces.add(nonce);

    return true;
  }

  // 定期清理过期的Nonce
  startCleanup(): void {
    this.cleanupInterval = setInterval(() => {
      // 清理5分钟前的Nonce记录
      this.usedNonces.clear();
    }, this.MAX_DELAY);
  }

  stopCleanup(): void {
    if (this.cleanupInterval) {
      clearInterval(this.cleanupInterval);
      this.cleanupInterval = null;
    }
  }
}
```
